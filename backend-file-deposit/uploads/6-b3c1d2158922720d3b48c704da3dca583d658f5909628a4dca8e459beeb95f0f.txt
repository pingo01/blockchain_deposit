存证记录表 file_cert（文件处理 + 存证查询模块核心）
CREATE TABLE `file_cert` (
  `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '存证记录ID',
  `user_id` INT NOT NULL COMMENT '关联上传者ID（外键关联user表）',
  `file_name` VARCHAR(255) NOT NULL COMMENT '文件名',
  `file_size` BIGINT NOT NULL COMMENT '文件大小（字节）',
  `file_type` VARCHAR(50) NOT NULL COMMENT '文件类型（如application/pdf、image/png）',
  `file_hash` VARCHAR(64) NOT NULL UNIQUE COMMENT '文件SHA256哈希值（防篡改核心）',
  `cert_id` VARCHAR(50) NOT NULL UNIQUE COMMENT '存证编号（对外查询用，格式：CERT_时间戳）',
  `chain_id` INT DEFAULT NULL COMMENT '关联区块链表ID（体现“上链”）',
  `upload_time` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '上传存证时间',
  -- 外键约束：确保存证记录归属正确用户
  FOREIGN KEY (`user_id`) REFERENCES `user`(`id`) ON DELETE CASCADE,
  -- 索引：加快查询速度
  INDEX `idx_cert_id` (`cert_id`),
  INDEX `idx_file_hash` (`file_hash`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='文件存证记录表';

区块链表 block_chain（区块链模拟核心模块核心）
CREATE TABLE `block_chain` (
  `id` INT AUTO_INCREMENT PRIMARY KEY COMMENT '区块ID（模拟“区块高度”，自增=高度）',
  `previous_hash` VARCHAR(64) NOT NULL COMMENT '上一区块的哈希值（构建哈希链的核心）',
  `current_hash` VARCHAR(64) NOT NULL UNIQUE COMMENT '当前区块的哈希值',
  `data` VARCHAR(2000) NOT NULL COMMENT '区块数据（存储存证编号列表，逗号分隔）',
  `timestamp` BIGINT NOT NULL COMMENT '区块生成时间戳（毫秒）',
  `nonce` INT DEFAULT 0 COMMENT '随机数（模拟挖矿，课程设计固定为0即可）'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='模拟区块链表';
